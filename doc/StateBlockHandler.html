def process
  statement.last.last.each do |stm|
    # regexps for expediency.  please rethink
    if stm.show =~ /(interface \w+,|\w+)\s*:(\w+),\s*\[([^]]*)\]\s*(=>\s*\[([^]]+)\])?/
      object = YARD::CodeObjects::MethodObject.new(namespace, $2)
      register(object)
      object.dynamic = true 
      object.parameters = [["[#{$3}]", nil], [$5, nil]]
      object.source = stm.source
      object.docstring = YARD::Docstring.new(stm.comments, object)
      parse_block(stm, :owner => object)        
    end
  end
end
